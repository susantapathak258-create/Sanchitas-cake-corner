SQL Setup

CREATE TABLE products (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(200), category VARCHAR(50), price DECIMAL(10,2), offer_price DECIMAL(10,2), emoji VARCHAR(10), photo_url VARCHAR(500), description TEXT, stock INT DEFAULT 100, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE orders (id VARCHAR(20) PRIMARY KEY, customer_name VARCHAR(200), phone VARCHAR(20), address TEXT, items JSON, total DECIMAL(10,2), coins_earned INT, status VARCHAR(50) DEFAULT 'pending', payment_status VARCHAR(50) DEFAULT 'pending', payment_method VARCHAR(50), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
CREATE TABLE customers (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(200), phone VARCHAR(20) UNIQUE, total_coins INT DEFAULT 0, total_spent DECIMAL(10,2) DEFAULT 0, order_count INT DEFAULT 0, membership VARCHAR(20) DEFAULT 'none', membership_expiry DATE, last_order TIMESTAMP);
CREATE TABLE reviews (id INT AUTO_INCREMENT PRIMARY KEY, product_id INT, product_name VARCHAR(200), reviewer_name VARCHAR(200), rating INT, review_text TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP);
